53 void StorageEngine::beginTask(QFutureInterface<QVariant> futureInterface, QThread *targetThread, StorageEngine::TaskType taskType, int metaTypeId, const QVariant &value)
54 {
55  try {
56  auto flags = QMetaType::typeFlags(metaTypeId);
57  auto metaObject = QMetaType::metaObjectForType(metaTypeId);
58  if((!flags.testFlag(QMetaType::PointerToQObject) &&
59  !flags.testFlag(QMetaType::IsGadget)) ||
60  !metaObject)
61  throw DataSyncException("You can only store QObjects or Q_GADGETs with QtDataSync!");
62 
63  auto userProp = metaObject->userProperty();
64  if(!userProp.isValid())
65  throw DataSyncException("To store a datatype, it requires a user property");
66 
67  switch (taskType) {
68  case Count:
69  count(futureInterface, targetThread, metaTypeId);
70  break;
71  case Keys:
72  keys(futureInterface, targetThread, metaTypeId);
73  break;
74  case LoadAll:
75  loadAll(futureInterface, targetThread, metaTypeId, value.toInt());
76  break;
77  case Load:
78  load(futureInterface, targetThread, metaTypeId, userProp.name(), value.toString());
79  break;
80  case Save:
81  save(futureInterface, targetThread, metaTypeId, userProp.name(), value);
82  break;
83  case Remove:
84  remove(futureInterface, targetThread, metaTypeId, userProp.name(), value.toString());
85  break;
86  case Search:
87  search(futureInterface, targetThread, metaTypeId, value.value<QPair<int, QString>>());
88  break;
89  default:
90  break;
91  }
92  } catch(QException &e) {
93  futureInterface.reportException(e);
94  futureInterface.reportFinished();
95  }
96 }

179 void StorageEngine::requestCompleted(quint64 id, const QJsonValue &result)
180 {
181  auto info = requestCache.take(id);
182 
183  if(info.isChangeControllerRequest) {
184  changeController->nextStage(true, result);
185  } else {
186  if(!result.isUndefined()) {
187  try {
188  auto obj = serializer->deserialize(result, info.convertMetaTypeId);
189  if(info.targetThread)
190  tryMoveToThread(obj, info.targetThread);
191  info.futureInterface.reportResult(obj);
192  } catch(QJsonSerializerException &e) {
193  info.futureInterface.reportException(e);
194  }
195  } else
196  info.futureInterface.reportResult(QVariant());
197 
198  info.futureInterface.reportFinished();
199  }
200 
201  if(info.isDeleteAction && !result.toBool())
202  return;
203 
204  if(info.changeAction) {
205  stateHolder->markLocalChanged(info.changeKey, info.changeState);
206  changeController->updateLocalStatus(info.changeKey, info.changeState);
207  }
208 
209  if(!info.notifyKey.first.isNull())
210  emit notifyChanged(QMetaType::type(info.notifyKey.first), info.notifyKey.second, info.isDeleteAction);
211 }

269 void StorageEngine::beginRemoteOperation(const ChangeController::ChangeOperation &operation)
270 {
271  auto metaTypeId = QMetaType::type(operation.key.first);
272  auto metaObject = QMetaType::metaObjectForType(metaTypeId);
273  if(!metaObject) {
274  qCWarning(LOG) << "Remote operation for invalid type"
275  << operation.key.first
276  << "requested!";
277  return;
278  }
279 
280  auto userProperty = QMetaType::metaObjectForType(metaTypeId)->userProperty();
281  switch (operation.operation) {
282  case ChangeController::Load:
283  remoteConnector->download(operation.key, userProperty.name());
284  break;
285  case ChangeController::Save:
286  remoteConnector->upload(operation.key, operation.writeObject, userProperty.name());
287  break;
288  case ChangeController::Remove:
289  remoteConnector->remove(operation.key, userProperty.name());
290  break;
291  case ChangeController::MarkUnchanged:
292  remoteConnector->markUnchanged(operation.key, userProperty.name());
293  break;
294  default:
295  Q_UNREACHABLE();
296  }
297 }

299 void StorageEngine::beginLocalOperation(const ChangeController::ChangeOperation &operation)
300 {
301  auto metaTypeId = QMetaType::type(operation.key.first);
302  auto metaObject = QMetaType::metaObjectForType(metaTypeId);
303  if(!metaObject) {
304  qCWarning(LOG) << "Local operation for invalid type"
305  << operation.key.first
306  << "requested!";
307  return;
308  }
309 
310  auto id = requestCounter++;
311  RequestInfo info(true);
312 
313  auto userProperty = metaObject->userProperty();
314  switch (operation.operation) {
315  case ChangeController::Load:
316  requestCache.insert(id, info);
317  localStore->load(id, operation.key, userProperty.name());
318  break;
319  case ChangeController::Save:
320  info.notifyKey = operation.key;
321  info.isDeleteAction = false;
322  info.changeAction = true;
323  info.changeKey = operation.key;
324  info.changeState = StateHolder::Unchanged;
325  requestCache.insert(id, info);
326  localStore->save(id, operation.key, operation.writeObject, userProperty.name());
327  break;
328  case ChangeController::Remove:
329  info.notifyKey = operation.key;
330  info.isDeleteAction = true;
331  info.changeAction = true;
332  info.changeKey = operation.key;
333  info.changeState = StateHolder::Unchanged;
334  requestCache.insert(id, info);
335  localStore->remove(id, operation.key, userProperty.name());
336  break;
337  case ChangeController::MarkUnchanged:
338  stateHolder->markLocalChanged(operation.key, StateHolder::Unchanged);
339  changeController->nextStage(true);
340  break;
341  default:
342  Q_UNREACHABLE();
343  }
344 }
