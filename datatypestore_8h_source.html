554 template <typename TType, typename TKey>
555 void CachingDataTypeStore<TType*, TKey>::evalDataChanged(int metaTypeId, const QString &key, bool wasDeleted)
556 {
557  if(metaTypeId == qMetaTypeId<TType*>()) {
558  auto rKey = toKey(key);
559  if(wasDeleted) {
560  auto data = _data.take(rKey);
561  if(data) {
562  emit dataChanged(key, QVariant());
563  data->deleteLater();
564  }
565  } else {
566  if(_data.contains(rKey)) {
567  _store->update(_data.value(rKey));
568  emit dataChanged(key, QVariant::fromValue(_data.value(rKey)));
569  } else {
570  auto data = _store->load<TType*>(key);
571  auto oldData = _data.take(rKey);
572  data->setParent(this);
573  _data.insert(rKey, data);
574  emit dataChanged(key, QVariant::fromValue(data));
575  if(oldData)
576  oldData->deleteLater();
577  }
578  }
579  }
580 }
