578 template <typename TType, typename TKey>
579 void CachingDataTypeStore<TType*, TKey>::evalDataChanged(int metaTypeId, const QString &key, bool wasDeleted)
580 {
581  if(metaTypeId == qMetaTypeId<TType*>()) {
582  auto rKey = toKey(key);
583  if(wasDeleted) {
584  auto data = _data.take(rKey);
585  if(data) {
586  emit dataChanged(key, QVariant());
587  data->deleteLater();
588  }
589  } else {
590  if(_data.contains(rKey)) {
591  _store->update(_data.value(rKey));
592  emit dataChanged(key, QVariant::fromValue(_data.value(rKey)));
593  } else {
594  auto data = _store->load<TType*>(key);
595  auto oldData = _data.take(rKey);
596  data->setParent(this);
597  _data.insert(rKey, data);
598  emit dataChanged(key, QVariant::fromValue(data));
599  if(oldData)
600  oldData->deleteLater();
601  }
602  }
603  }
604 }
