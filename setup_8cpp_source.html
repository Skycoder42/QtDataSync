149 void Setup::create(const QString &name)
150 {
151  QMutexLocker _(&SetupPrivate::setupMutex);
152  if(SetupPrivate::engines.contains(name))
153  throw SetupExistsException(name);
154 
155  QDir storageDir = QStandardPaths::writableLocation(QStandardPaths::AppLocalDataLocation);
156  if(!storageDir.cd(d->localDir)) {
157  storageDir.mkpath(d->localDir);
158  storageDir.cd(d->localDir);
159  QFile::setPermissions(storageDir.absolutePath(),
160  QFileDevice::ReadUser | QFileDevice::WriteUser | QFileDevice::ExeUser);
161  }
162 
163  auto lockFile = new QLockFile(storageDir.absoluteFilePath(QStringLiteral(".lock")));
164  if(!lockFile->tryLock())
165  throw SetupLockedException(name);
166 
167  auto defaults = new Defaults(name, storageDir, d->properties);
168 
169  auto engine = new StorageEngine(defaults,
170  d->serializer.take(),
171  d->localStore.take(),
172  d->stateHolder.take(),
173  d->remoteConnector.take(),
174  d->dataMerger.take(),
175  d->encryptor.take());
176 
177  auto thread = new QThread();
178  engine->moveToThread(thread);
179  QObject::connect(thread, &QThread::started,
180  engine, &StorageEngine::initialize);
181  QObject::connect(thread, &QThread::finished,
182  engine, &StorageEngine::deleteLater);
183  QObject::connect(engine, &StorageEngine::destroyed, qApp, [=](){
184  auto t = storageDir.path();
185  lockFile->unlock();
186  delete lockFile;
187  }, Qt::DirectConnection);
188  QObject::connect(thread, &QThread::finished, thread, [=](){
189  QMutexLocker _(&SetupPrivate::setupMutex);
190  SetupPrivate::engines.remove(name);
191  thread->deleteLater();
192  }, Qt::QueuedConnection);
193  thread->start();
194  SetupPrivate::engines.insert(name, {thread, engine});
195 }

221 void SetupPrivate::cleanupHandler()
222 {
223  QMutexLocker _(&setupMutex);
224  foreach (auto info, engines) {
225  if(info.second) {
226  QMetaObject::invokeMethod(info.second, "finalize", Qt::QueuedConnection);
227  info.second = nullptr;
