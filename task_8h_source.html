207 template<typename T>
208 T UpdateTask<T>::interalGet(QSharedPointer<UpdateData> data, QVariant result)
209 {
210  data->mutex->lock();
211  if(!data->updated) {
212  try {
213  auto newResult = result.template value<T>();
214  auto oldMeta = data->data->metaObject();
215  auto newMeta = newResult->metaObject();
216 
217  //verify objects match
218  if(!newMeta->inherits(oldMeta))
219  throw DataSyncException("loadInto: Loaded class type does not match the one to load into!");
220  auto user = oldMeta->userProperty();
221  if(user.read(newResult).toString() != user.read(data->data).toString())
222  throw DataSyncException("loadInto: The id of the loaded data does not match the one to be updated!");
223 
224  //pass on properties
225  for(auto i = 1; i < oldMeta->propertyCount(); i++) {//skip object name
226  auto prop = oldMeta->property(i);
227  if(prop.isStored())
228  prop.write(data->data, prop.read(newResult));
229  }
230  foreach(auto dynProp, newResult->dynamicPropertyNames())
231  data->data->setProperty(dynProp, newResult->property(dynProp));
232 
233  data->updated = true;
234  newResult->deleteLater();
235  } catch(...) {
236  data->mutex->unlock();
237  throw;
238  }
239  }
240  data->mutex->unlock();
241 
242  return data->data;
243 }
