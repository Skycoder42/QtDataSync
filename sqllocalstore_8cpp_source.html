195 {
196  TYPE_DIR(id, key.first)
197 
198  //check if the file exists
199  QSqlQuery existQuery(database);
200  existQuery.prepare(QStringLiteral("SELECT File FROM DataIndex WHERE Type = ? AND Key = ?"));
201  existQuery.addBindValue(key.first);
202  existQuery.addBindValue(key.second);
203  EXEC_QUERY(existQuery);
204 
205  auto needUpdate = false;
206  QScopedPointer<QFile> file;
207  if(existQuery.first()) {
208  file.reset(new QFile(tableDir.absoluteFilePath(existQuery.value(0).toString() + QStringLiteral(".dat"))));
209  file->open(QIODevice::WriteOnly);
210  } else {
211  auto fileName = QString::fromUtf8(QUuid::createUuid().toRfc4122().toHex());
212  fileName = tableDir.absoluteFilePath(QStringLiteral("%1XXXXXX.dat")).arg(fileName);
213  auto tFile = new QTemporaryFile(fileName);
214  tFile->setAutoRemove(false);
215  tFile->open();
216  file.reset(tFile);
217  needUpdate = true;
218  }
219 
220  //save the file
221  QJsonDocument doc(object);
222  file->write(doc.toBinaryData());
223  file->close();
224  if(file->error() != QFile::NoError) {
225  emit requestFailed(id, QStringLiteral("Failed to write data to file \"%1\" with error: %2")
226  .arg(file->fileName())
227  .arg(file->errorString()));
228  return;
229  }
230 
231  //save key in database
232  if(needUpdate) {
233  QSqlQuery insertQuery(database);
234  insertQuery.prepare(QStringLiteral("INSERT OR REPLACE INTO DataIndex (Type, Key, File) VALUES(?, ?, ?)"));
235  insertQuery.addBindValue(key.first);
236  insertQuery.addBindValue(key.second);
237  insertQuery.addBindValue(tableDir.relativeFilePath(QFileInfo(file->fileName()).completeBaseName()));
238  EXEC_QUERY(insertQuery);
239  }
240 
241  emit requestCompleted(id, QJsonValue::Undefined);
242 }
