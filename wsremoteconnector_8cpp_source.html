57 void WsRemoteConnector::reconnect()
58 {
59  if(state == Connecting ||
60  state == Closing)
61  return;
62 
63  if(socket) {
64  state = Closing;
65  connect(socket, &QWebSocket::destroyed,
66  this, &WsRemoteConnector::reconnect,
67  Qt::QueuedConnection);
68  socket->close();
69  } else {
70  emit clearAuthenticationError();
71  state = Connecting;
72  settings->sync();
73 
74  auto remoteUrl = settings->value(keyRemoteUrl).toUrl();
75  if(!remoteUrl.isValid()) {
76  state = Disconnected;
77  remoteStateChanged(RemoteDisconnected);
78  return;
79  }
80 
81  socket = new QWebSocket(settings->value(keySharedSecret, QStringLiteral("QtDataSync")).toString(),
82  QWebSocketProtocol::VersionLatest,
83  this);
84 
85  if(!settings->value(keyVerifyPeer, true).toBool()) {
86  auto conf = socket->sslConfiguration();
87  conf.setPeerVerifyMode(QSslSocket::VerifyNone);
88  socket->setSslConfiguration(conf);
89  }
90 
91  connect(socket, &QWebSocket::connected,
92  this, &WsRemoteConnector::connected);
93  connect(socket, &QWebSocket::binaryMessageReceived,
94  this, &WsRemoteConnector::binaryMessageReceived);
95  connect(socket, QOverload<QAbstractSocket::SocketError>::of(&QWebSocket::error),
96  this, &WsRemoteConnector::error);
97  connect(socket, &QWebSocket::sslErrors,
98  this, &WsRemoteConnector::sslErrors);
99  connect(socket, &QWebSocket::disconnected,
100  this, &WsRemoteConnector::disconnected,
101  Qt::QueuedConnection);
102 
103  QNetworkRequest request(remoteUrl);
104  request.setAttribute(QNetworkRequest::FollowRedirectsAttribute, true);
105  request.setAttribute(QNetworkRequest::HttpPipeliningAllowedAttribute, true);
106  request.setAttribute(QNetworkRequest::SpdyAllowedAttribute, true);
107  request.setAttribute(QNetworkRequest::HTTP2AllowedAttribute, true);
108 
109  settings->beginGroup(keyHeadersGroup);
110  auto keys = settings->childKeys();
111  foreach(auto key, keys)
112  request.setRawHeader(key.toUtf8(), settings->value(key).toByteArray());
113  settings->endGroup();
114 
115  socket->open(request);
116  }
117 }

119 void WsRemoteConnector::reloadRemoteState()
120 {
121  switch (state) {
122  case Disconnected:
123  reconnect();
124  break;
125  case Idle:
126  remoteStateChanged(RemoteLoadingState);
127  if(needResync)
128  emit performLocalReset(false);//direct connected, thus "inline"
129  state = Reloading;
130  sendCommand("loadChanges", needResync);
131  needResync = false;
132  break;
133  case Operating:
134  retry();
135  break;
136  default:
137  break;
138  }
139 }

249 void WsRemoteConnector::disconnected()
250 {
251  operationTimer->stop();//just to make shure
252 
253  if(state == Operating)
254  emit operationFailed(QStringLiteral("Connection closed"));
255 
256  if(state != Closing) {
257  auto delta = retry();
258  if(state == Connecting) {
259  qCWarning(LOG) << "Unable to connect to server. Retrying in"
260  << delta / 1000
261  << "seconds";
262  } else {
263  qCWarning(LOG) << "Unexpected disconnect from server with exit code"
264  << socket->closeCode()
265  << ":"
266  << socket->closeReason()
267  << ". Retrying in"
268  << delta / 1000
269  << "seconds";
270  }
271  }
272 
273  state = Disconnected;
274  socket->deleteLater();
275  socket = nullptr;
276 
277  //always "disable" remote for the state changer
278  emit remoteStateChanged(RemoteDisconnected);
279 }

281 void WsRemoteConnector::binaryMessageReceived(const QByteArray &message)
282 {
283  operationTimer->stop();
284 
285  QJsonParseError error;
286  auto doc = QJsonDocument::fromJson(message, &error);
287  if(error.error != QJsonParseError::NoError) {
288  qCWarning(LOG) << "Invalid data received. Parser error:"
289  << error.errorString();
290  return;
291  }
292 
293  auto obj = doc.object();
294  auto data = obj[QStringLiteral("data")];
295  if(obj[QStringLiteral("command")] == QStringLiteral("identified"))
296  identified(data.toObject());
297  else if(obj[QStringLiteral("command")] == QStringLiteral("identifyFailed"))
298  identifyFailed();
299  else if(obj[QStringLiteral("command")] == QStringLiteral("changeState"))
300  changeState(data.toObject());
301  else if(obj[QStringLiteral("command")] == QStringLiteral("notifyChanged"))
302  notifyChanged(data.toObject());
303  else if(obj[QStringLiteral("command")] == QStringLiteral("completed"))
304  completed(data.toObject());
305  else {
306  qCWarning(LOG) << "Unkown command received:"
307  << obj[QStringLiteral("command")].toString();
308  }
309 }

363 void WsRemoteConnector::changeState(const QJsonObject &data)
364 {
365  if(data[QStringLiteral("success")].toBool()) {
366  //reset retry & need resync
367  retryIndex = 0;
368 
369  StateHolder::ChangeHash changeState;
370  foreach(auto value, data[QLatin1String("data")].toArray()) {
371  auto obj = value.toObject();
372  ObjectKey key;
373  key.first = obj[QStringLiteral("type")].toString().toUtf8();
374  key.second = obj[QStringLiteral("key")].toString();
375  changeState.insert(key, obj[QStringLiteral("changed")].toBool() ? StateHolder::Changed : StateHolder::Deleted);
376  }
377  emit remoteStateChanged(RemoteReady, changeState);
378  } else {
379  auto delta = retry();
380  qCWarning(LOG) << "Failed to load state with error:"
381  << data[QStringLiteral("error")].toString()
382  << ". Retrying in"
383  << delta / 1000
384  << "seconds";
385  }
386  state = Idle;
387 }
