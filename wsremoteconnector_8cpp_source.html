78 void WsRemoteConnector::reconnect()
79 {
80  if(state == Connecting ||
81  state == Closing)
82  return;
83 
84  if(socket) {
85  state = Closing;
86  connect(socket, &QWebSocket::destroyed,
87  this, &WsRemoteConnector::reconnect,
88  Qt::QueuedConnection);
89  socket->close();
90  } else {
91  emit clearAuthenticationError();
92  state = Connecting;
93  settings->sync();
94  emit remoteStateChanged(RemoteConnecting);
95 
96  auto remoteUrl = settings->value(keyRemoteUrl).toUrl();
97  if(!remoteUrl.isValid() || !settings->value(keyRemoteEnabled, true).toBool()) {
98  state = Disconnected;
99  emit remoteStateChanged(RemoteDisconnected);
100  return;
101  }
102 
103  socket = new QWebSocket(settings->value(keySharedSecret, QStringLiteral("QtDataSync")).toString(),
104  QWebSocketProtocol::VersionLatest,
105  this);
106 
107  if(!settings->value(keyVerifyPeer, true).toBool()) {
108  auto conf = socket->sslConfiguration();
109  conf.setPeerVerifyMode(QSslSocket::VerifyNone);
110  socket->setSslConfiguration(conf);
111  }
112 
113  connect(socket, &QWebSocket::connected,
114  this, &WsRemoteConnector::connected);
115  connect(socket, &QWebSocket::binaryMessageReceived,
116  this, &WsRemoteConnector::binaryMessageReceived);
117  connect(socket, QOverload<QAbstractSocket::SocketError>::of(&QWebSocket::error),
118  this, &WsRemoteConnector::error);
119  connect(socket, &QWebSocket::sslErrors,
120  this, &WsRemoteConnector::sslErrors);
121  connect(socket, &QWebSocket::disconnected,
122  this, &WsRemoteConnector::disconnected,
123  Qt::QueuedConnection);
124 
125  QNetworkRequest request(remoteUrl);
126  request.setAttribute(QNetworkRequest::FollowRedirectsAttribute, true);
127  request.setAttribute(QNetworkRequest::HttpPipeliningAllowedAttribute, true);
128  request.setAttribute(QNetworkRequest::SpdyAllowedAttribute, true);
129  request.setAttribute(QNetworkRequest::HTTP2AllowedAttribute, true);
130 
131  settings->beginGroup(keyHeadersGroup);
132  auto keys = settings->childKeys();
133  foreach(auto key, keys)
134  request.setRawHeader(key.toUtf8(), settings->value(key).toByteArray());
135  settings->endGroup();
136 
137  socket->open(request);
138  }
139 }

141 void WsRemoteConnector::reloadRemoteState()
142 {
143  switch (state) {
144  case Disconnected:
145  reconnect();
146  break;
147  case Idle:
148  emit remoteStateChanged(RemoteLoadingState);
149  if(needResync)
150  emit performLocalReset(false);//direct connected, thus "inline"
151  state = Reloading;
152  sendCommand("loadChanges", needResync);
153  needResync = false;
154  break;
155  case Operating:
156  retry();
157  break;
158  default:
159  break;
160  }
161 }

284 void WsRemoteConnector::disconnected()
285 {
286  operationTimer->stop();//just to make shure
287 
288  if(state == Operating)
289  emit operationFailed(QStringLiteral("Connection closed"));
290 
291  if(state != Closing) {
292  auto delta = retry();
293  if(state == Connecting) {
294  qCWarning(LOG) << "Unable to connect to server. Retrying in"
295  << delta / 1000
296  << "seconds";
297  } else {
298  qCWarning(LOG) << "Unexpected disconnect from server with exit code"
299  << socket->closeCode()
300  << ":"
301  << socket->closeReason()
302  << ". Retrying in"
303  << delta / 1000
304  << "seconds";
305  }
306  }
307 
308  state = Disconnected;
309  socket->deleteLater();
310  socket = nullptr;
311 
312  //always "disable" remote for the state changer
313  emit remoteStateChanged(RemoteDisconnected);
314 }

316 void WsRemoteConnector::binaryMessageReceived(const QByteArray &message)
317 {
318  operationTimer->stop();
319 
320  QJsonParseError error;
321  auto doc = QJsonDocument::fromJson(message, &error);
322  if(error.error != QJsonParseError::NoError) {
323  qCWarning(LOG) << "Invalid data received. Parser error:"
324  << error.errorString();
325  return;
326  }
327 
328  auto obj = doc.object();
329  auto data = obj[QStringLiteral("data")];
330  if(obj[QStringLiteral("command")] == QStringLiteral("identified"))
331  identified(data.toObject());
332  else if(obj[QStringLiteral("command")] == QStringLiteral("identifyFailed"))
333  identifyFailed();
334  else if(obj[QStringLiteral("command")] == QStringLiteral("changeState"))
335  changeState(data.toObject());
336  else if(obj[QStringLiteral("command")] == QStringLiteral("notifyChanged"))
337  notifyChanged(data.toObject());
338  else if(obj[QStringLiteral("command")] == QStringLiteral("completed"))
339  completed(data.toObject());
340  else {
341  qCWarning(LOG) << "Unkown command received:"
342  << obj[QStringLiteral("command")].toString();
343  }
344 }

401 void WsRemoteConnector::changeState(const QJsonObject &data)
402 {
403  if(data[QStringLiteral("success")].toBool()) {
404  //reset retry & need resync
405  retryIndex = 0;
406 
407  StateHolder::ChangeHash changeState;
408  foreach(auto value, data[QLatin1String("data")].toArray()) {
409  auto obj = value.toObject();
410  ObjectKey key;
411  key.first = obj[QStringLiteral("type")].toString().toUtf8();
412  key.second = obj[QStringLiteral("key")].toString();
413  changeState.insert(key, obj[QStringLiteral("changed")].toBool() ? StateHolder::Changed : StateHolder::Deleted);
414  }
415  emit remoteStateChanged(RemoteReady, changeState);
416  } else {
417  auto delta = retry();
418  qCWarning(LOG) << "Failed to load state with error:"
419  << data[QStringLiteral("error")].toString()
420  << ". Retrying in"

422  << "seconds";
423  }
