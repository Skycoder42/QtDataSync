322 template <typename TType, typename TKey>
323 void CachingDataStore<TType*, TKey>::evalDataChanged(int metaTypeId, const QString &key, bool wasDeleted)
324 {
325  if(metaTypeId == qMetaTypeId<TType*>()) {
326  auto rKey = toKey(key);
327  if(wasDeleted) {
328  auto data = _data.take(rKey);
329  if(data) {
330  emit dataChanged(key, QVariant());
331  data->deleteLater();
332  }
333  } else {
334  if(_data.contains(rKey)) {
335  _store->loadInto<TType*>(key, _data.value(rKey)).onResult(this, [=](TType* data) {
336  auto oldData = _data.value(rKey, nullptr);
337  if(oldData == data)
338  emit dataChanged(key, QVariant::fromValue(data));
339  else {
340  data->setParent(this);
341  _data.insert(rKey, data);
342  emit dataChanged(key, QVariant::fromValue(data));
343  if(oldData)
344  oldData->deleteLater();
345  }
346  });
347  } else {
348  _store->load<TType*>(key).onResult(this, [=](TType* data) {
349  auto oldData = _data.take(rKey);
350  data->setParent(this);
351  _data.insert(rKey, data);
352  emit dataChanged(key, QVariant::fromValue(data));
353  if(oldData)
354  oldData->deleteLater();
355  });
356  }
