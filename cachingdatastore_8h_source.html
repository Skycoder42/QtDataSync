286 void CachingDataStore<TType*, TKey>::save(TType *value)
287 {
288  auto userProp = TType::staticMetaObject.userProperty();
289  auto keyVariant = userProp.read(value);
290  auto key = keyVariant.template value<TKey>();
291  auto keyString = keyVariant.toString();
292 
293  auto data = _data.value(key, nullptr);
294  if(data == value) {
295  _store->save(value);
296  emit dataChanged(keyString, QVariant::fromValue(value));
297  } else {
298  value->setParent(this);
299  _data.insert(key, value);
300  _store->save(value);
301  emit dataChanged(keyString, QVariant::fromValue(value));
302  if(data)
303  data->deleteLater();
304  }
305 }

325 template <typename TType, typename TKey>
326 void CachingDataStore<TType*, TKey>::evalDataChanged(int metaTypeId, const QString &key, bool wasDeleted)
327 {
328  if(metaTypeId == qMetaTypeId<TType*>()) {
329  auto rKey = toKey(key);
330  if(wasDeleted) {
331  auto data = _data.take(rKey);
332  if(data) {
333  emit dataChanged(key, QVariant());
334  data->deleteLater();
335  }
336  } else {
337  if(_data.contains(rKey)) {
338  _store->loadInto<TType*>(key, _data.value(rKey)).onResult(this, [=](TType* data) {
339  auto oldData = _data.value(rKey, nullptr);
340  if(oldData == data)
341  emit dataChanged(key, QVariant::fromValue(data));
342  else {
343  data->setParent(this);
344  _data.insert(rKey, data);
345  emit dataChanged(key, QVariant::fromValue(data));
346  if(oldData)
347  oldData->deleteLater();
348  }
349  });
350  } else {
351  _store->load<TType*>(key).onResult(this, [=](TType* data) {
352  auto oldData = _data.take(rKey);
353  data->setParent(this);
354  _data.insert(rKey, data);
355  emit dataChanged(key, QVariant::fromValue(data));
356  if(oldData)
357  oldData->deleteLater();
358  });
359  }
360  }
361  }
